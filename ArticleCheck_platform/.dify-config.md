# ==========================================================
# Article Check — Dify 平台接入说明（当前真实架构）
# ==========================================================
#
# 重要说明：
# - 当前仓库已经支持由 FastAPI 后端代理 Dify Service API。
# - 当前推荐的落地方式是：
#   1) 前端仍作为 WebDemo 部署
#   2) FastAPI 后端统一调用 Dify
#   3) Dify API Key 仅保存在服务端容器环境变量中
#
# ----------------------------------------------------------
# 一、Dify 自身部署信息
# ----------------------------------------------------------
#
# Dify API 基础地址（示例：Docker 自托管默认走 Nginx）
DIFY_BASE_URL=http://host.docker.internal/v1
#
# Dify 应用 API Key（在 Dify Studio 中创建应用后获取）
DIFY_API_KEY=app-your-dify-api-key-here
#
# ----------------------------------------------------------
# 二、Article Check 服务地址
# ----------------------------------------------------------
#
# 若 Article Check 与 Dify 在同一台宿主机上，且 Article Check 以 Docker Compose 部署：
# 对外入口通常为：
ARTICLE_CHECK_BASE_URL=http://host.docker.internal:3000
#
# 若 Dify 容器需要直接调用 Article Check API，可使用：
ARTICLE_CHECK_API_BASE_URL=http://host.docker.internal:3000/api
#
# ----------------------------------------------------------
# 三、当前支持的 Dify 应用模式
# ----------------------------------------------------------
#
# 模式 A：chat
# - 设置：DIFY_APP_TYPE=chat
# - 后端调用 Dify `/chat-messages`
# - 适合项目方给出的聊天型示例 yml
#
# 模式 B：workflow
# - 设置：DIFY_APP_TYPE=workflow
# - 后端调用 Dify `/workflows/run`
# - 适合输出结构化结果的工作流应用
#
# ----------------------------------------------------------
# 四、后端容器环境变量样例
# ----------------------------------------------------------
#
# ARTICLE_CHECK_AI_PROVIDER=dify
# DIFY_BASE_URL=http://host.docker.internal/v1
# DIFY_API_KEY=app-your-dify-api-key-here
# DIFY_APP_TYPE=chat
# DIFY_RESPONSE_MODE=blocking
#
# 若使用 Workflow，可继续配置：
# DIFY_WORKFLOW_QUERY_KEY=query
# DIFY_INPUTS_JSON={}
#
# ----------------------------------------------------------
# 五、当前实现要点
# ----------------------------------------------------------
#
# 1. 前端不直接调用 Dify，避免泄露 API Key
# 2. 论文审查页、报告页、Evidence 联动仍保留在本项目中
# 3. WebDemo 可通过 `auth.js` 接入项目方认证服务
# 4. Docker 部署时建议使用 `.env.docker` 作为基础模板
